*Set

  =A set is an unordered collection that does not allow duplicates
  
  Set<String> vehicles={'car','bus','bike'];
  
  result={'car','bike'}
  
  //unique values only,no guaranteed order,no index access
  
1.Adding elements

void main() {
  // STRING
  String name = 'Flutter Developer';

  print(name.length);              // Length of string
  print(name.toUpperCase());       // Uppercase
  print(name.toLowerCase());       // Lowercase
  print(name.contains('Dev'));     // true
  print(name.replaceAll('Flutter', 'Dart'));
  print(name.split(' '));          // Convert to List

  print('-------------------');

  // INT
  int age = 25;

  print(age.isEven);               // true/false
  print(age.isOdd);                // true/false
  print(age.abs());                // Absolute value
  print(age.compareTo(30));        // -1, 0, or 1
  print(age.toString());           // Convert to String

  print('-------------------');

  // DOUBLE
  double price = 199.99;

  print(price.round());            // 200
  print(price.floor());            // 199
  print(price.ceil());             // 200
  print(price.toStringAsFixed(1)); // 200.0

  print('-------------------');

  // LIST
  List<String> fruits = ['Apple', 'Banana', 'Orange'];

  fruits.add('Mango');             // Add item
  fruits.remove('Banana');         // Remove item
  print(fruits.length);            // Count
  print(fruits.first);             // First item
  print(fruits.last);              // Last item
  print(fruits.contains('Apple')); // true
  fruits.sort();                   // Sort list

  print(fruits);

  print('-------------------');

  // SET
  Set<String> vehicles = {'Car', 'Bus', 'Bike'};

  vehicles.add('Truck');           // Add
  vehicles.add('Car');             // Duplicate ignored
  vehicles.remove('Bus');          // Remove
  print(vehicles.contains('Bike'));
  print(vehicles.length);

  print(vehicles);

  print('-------------------');

  // BOOL
  bool isLoggedIn = true;

  print(isLoggedIn);               // true
  print(!isLoggedIn);              // false (NOT operator)

  print('-------------------');

  // MAP
  Map<String, dynamic> picture = {
    'url': 'photo.png',
    'width': 500,
    'height': 300,
    'isPublic': true,
  };

  print(picture['url']);           // Access value
  picture['size'] = '500KB';       // Add new key
  picture.remove('height');        // Remove key
  print(picture.keys);             // All keys
  print(picture.values);           // All values
  print(picture.containsKey('url'));
  print(picture.containsValue(true));

  print(picture);

  print('-------------------');

  // VAR
  var money = 1500;                // int
  print(money.runtimeType);

  money = 2000;                    // Still int
  print(money);
}
